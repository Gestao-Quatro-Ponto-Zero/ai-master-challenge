## Modelo de banco sugerido

Use SQLite com chaves substitutas (`id`) nas dimensões e mantenha os identificadores originais únicos.

Tabelas:

- `accounts`
  - `id` (PK)
  - `name` (UNIQUE)  <- `account`
  - `sector`
  - `year_established`
  - `revenue_musd`
  - `employees`
  - `office_location`
  - `parent_account_id` (FK para `accounts.id`, nullable)

- `products`
  - `id` (PK)
  - `name` (UNIQUE) <- `product`
  - `series`
  - `sales_price`

- `sales_teams`
  - `id` (PK)
  - `sales_agent` (UNIQUE)
  - `manager`
  - `regional_office`

- `opportunities` (fato)
  - `id` (PK)
  - `opportunity_id` (UNIQUE)
  - `sales_agent_id` (FK -> `sales_teams.id`)
  - `product_id` (FK -> `products.id`)
  - `account_id` (FK -> `accounts.id`, nullable se vier vazio)
  - `deal_stage` (CHECK: `Prospecting|Engaging|Won|Lost`)
  - `engage_date` (DATE)
  - `close_date` (DATE, nullable)
  - `close_value` (NUMERIC, nullable)

- `ingestion_issues` (recomendado)
  - para registrar linhas com erro de chave/parse sem perder rastreabilidade.

## ETL: ordem correta

1. Carregar CSV em DataFrames.
2. Normalizar strings (`strip`, espaços duplos, case consistente).
3. Resolver mapeamentos conhecidos:
   - `GTXPro -> GTX Pro` (e outros que aparecerem)
4. Converter tipos:
   - datas (`engage_date`, `close_date`)
   - numéricos (`revenue`, `sales_price`, `close_value`)
5. Regras de qualidade:
   - `Won/Lost` devem ter `close_date`
   - `Engaging/Prospecting` podem ter `close_date = null`
   - `close_value = 0` para `Lost` pode ser aceito, mas registre regra
6. Carga nas dimensões (`accounts`, `products`, `sales_teams`) com deduplicação.
7. Resolver FKs e carregar `opportunities`.
8. Registrar inválidos em `ingestion_issues`.

```python
# db/db_init.py
import csv
import sqlite3
from pathlib import Path
from typing import Optional


BASE_DIR = Path(__file__).resolve().parent.parent
DATA_DIR = BASE_DIR / "data"
DB_PATH = BASE_DIR / "db" / "sales.sqlite3"

PRODUCT_ALIASES = {
    "GTXPro": "GTX Pro",
}


def to_int(value: str) -> Optional[int]:
    value = (value or "").strip()
    if not value:
        return None
    return int(value)


def to_float(value: str) -> Optional[float]:
    value = (value or "").strip()
    if not value:
        return None
    return float(value)


def normalize_product(value: str) -> Optional[str]:
    value = (value or "").strip()
    if not value:
        return None
    return PRODUCT_ALIASES.get(value, value)


def create_tables(conn: sqlite3.Connection) -> None:
    conn.execute("PRAGMA foreign_keys = ON;")

    conn.execute(
        """
        CREATE TABLE IF NOT EXISTS accounts (
            account TEXT PRIMARY KEY,
            sector TEXT,
            year_established INTEGER,
            revenue REAL,
            employees INTEGER,
            office_location TEXT,
            subsidiary_of TEXT
        );
        """
    )

    conn.execute(
        """
        CREATE TABLE IF NOT EXISTS products (
            product TEXT PRIMARY KEY,
            series TEXT,
            sales_price REAL
        );
        """
    )

    conn.execute(
        """
        CREATE TABLE IF NOT EXISTS sales_teams (
            sales_agent TEXT PRIMARY KEY,
            manager TEXT,
            regional_office TEXT
        );
        """
    )

    conn.execute(
        """
        CREATE TABLE IF NOT EXISTS sales_pipeline (
            opportunity_id TEXT PRIMARY KEY,
            sales_agent TEXT,
            product TEXT,
            account TEXT,
            deal_stage TEXT,
            engage_date TEXT,
            close_date TEXT,
            close_value REAL,
            FOREIGN KEY (sales_agent) REFERENCES sales_teams (sales_agent),
            FOREIGN KEY (product) REFERENCES products (product),
            FOREIGN KEY (account) REFERENCES accounts (account)
        );
        """
    )


def import_accounts(conn: sqlite3.Connection) -> None:
    path = DATA_DIR / "accounts.csv"
    with path.open(newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        rows = []
        for r in reader:
            rows.append(
                (
                    (r["account"] or "").strip(),
                    (r["sector"] or "").strip() or None,
                    to_int(r["year_established"]),
                    to_float(r["revenue"]),
                    to_int(r["employees"]),
                    (r["office_location"] or "").strip() or None,
                    (r["subsidiary_of"] or "").strip() or None,
                )
            )

    conn.executemany(
        """
        INSERT OR IGNORE INTO accounts (
            account, sector, year_established, revenue, employees, office_location, subsidiary_of
        ) VALUES (?, ?, ?, ?, ?, ?, ?);
        """,
        rows,
    )


def import_products(conn: sqlite3.Connection) -> None:
    path = DATA_DIR / "products.csv"
    with path.open(newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        rows = []
        for r in reader:
            rows.append(
                (
                    (r["product"] or "").strip(),
                    (r["series"] or "").strip() or None,
                    to_float(r["sales_price"]),
                )
            )

    conn.executemany(
        """
        INSERT OR IGNORE INTO products (product, series, sales_price)
        VALUES (?, ?, ?);
        """,
        rows,
    )


def import_sales_teams(conn: sqlite3.Connection) -> None:
    path = DATA_DIR / "sales_teams.csv"
    with path.open(newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        rows = []
        for r in reader:
            rows.append(
                (
                    (r["sales_agent"] or "").strip(),
                    (r["manager"] or "").strip() or None,
                    (r["regional_office"] or "").strip() or None,
                )
            )

    conn.executemany(
        """
        INSERT OR IGNORE INTO sales_teams (sales_agent, manager, regional_office)
        VALUES (?, ?, ?);
        """,
        rows,
    )


def import_sales_pipeline(conn: sqlite3.Connection) -> None:
    path = DATA_DIR / "sales_pipeline.csv"
    with path.open(newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        rows = []
        for r in reader:
            product = normalize_product(r["product"])
            account = (r["account"] or "").strip() or None
            close_date = (r["close_date"] or "").strip() or None
            close_value = to_float(r["close_value"])

            rows.append(
                (
                    (r["opportunity_id"] or "").strip(),
                    (r["sales_agent"] or "").strip() or None,
                    product,
                    account,
                    (r["deal_stage"] or "").strip() or None,
                    (r["engage_date"] or "").strip() or None,
                    close_date,
                    close_value,
                )
            )

    conn.executemany(
        """
        INSERT OR IGNORE INTO sales_pipeline (
            opportunity_id, sales_agent, product, account, deal_stage, engage_date, close_date, close_value
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?);
        """,
        rows,
    )


def init_db() -> None:
    DB_PATH.parent.mkdir(parents=True, exist_ok=True)

    with sqlite3.connect(DB_PATH) as conn:
        create_tables(conn)
        import_accounts(conn)
        import_products(conn)
        import_sales_teams(conn)
        import_sales_pipeline(conn)
        conn.commit()

    print(f"Banco inicializado com sucesso em: {DB_PATH}")


if __name__ == "__main__":
    init_db()
```

```python
import os
from datetime import datetime

import pandas as pd
import requests
import streamlit as st

API_BASE = os.getenv("API_BASE", "http://127.0.0.1:8000")

st.set_page_config(page_title="g4sales Priorities", layout="wide")
st.title("g4sales - Priority Desk")
st.caption("Veja o pipeline e saiba onde focar hoje.")

@st.cache_data(ttl=60)
def fetch_opportunities(limit=2000, deal_stage=None, sales_agent=None, product=None):
    params = {"limit": limit, "offset": 0}
    if deal_stage:
        params["deal_stage"] = deal_stage
    if sales_agent:
        params["sales_agent"] = sales_agent
    if product:
        params["product"] = product
    r = requests.get(f"{API_BASE}/opportunities", params=params, timeout=10)
    r.raise_for_status()
    return pd.DataFrame(r.json())

def days_since(date_str):
    if not date_str:
        return 999
    d = datetime.strptime(date_str, "%Y-%m-%d").date()
    return (datetime.utcnow().date() - d).days

def score_row(row):
    # Base por estágio
    stage_weight = {
        "Engaging": 40,
        "Prospecting": 25,
        "Won": -999,
        "Lost": -999,
    }
    score = stage_weight.get(row.get("deal_stage"), 0)

    # Valor potencial (se houver close_value)
    close_value = row.get("close_value") or 0
    score += min(close_value / 150, 20)

    # Urgência por tempo em pipeline
    age = row.get("days_in_pipeline", 999)
    if age > 30:
        score += 15
    elif age > 14:
        score += 8

    # Penalidade por dados incompletos
    if not row.get("account"):
        score -= 8
    if not row.get("product"):
        score -= 8

    return round(score, 2)

def explain_row(row):
    reasons = []
    if row["deal_stage"] == "Engaging":
        reasons.append("deal já em estágio avançado (Engaging)")
    if row["days_in_pipeline"] > 30:
        reasons.append("está esfriando no pipeline")
    if (row.get("close_value") or 0) > 2000:
        reasons.append("potencial de valor relevante")
    if not reasons:
        reasons.append("bom equilíbrio entre estágio e urgência")
    return "; ".join(reasons)

def next_action(row):
    stage = row.get("deal_stage")
    age = row.get("days_in_pipeline", 999)
    value = row.get("close_value") or 0

    if stage == "Engaging" and age > 21:
        return "Ligar hoje e agendar reunião de fechamento em até 48h."
    if stage == "Engaging" and value > 3000:
        return "Enviar proposta + envolver manager para destravar decisão."
    if stage == "Prospecting" and age > 14:
        return "Reativar contato com e-mail objetivo e CTA para reunião curta."
    return "Fazer follow-up hoje e definir próximo passo com data."

# Filtros
c1, c2, c3 = st.columns(3)
with c1:
    deal_stage = st.selectbox("Estágio", ["", "Prospecting", "Engaging"])
with c2:
    sales_agent = st.text_input("Sales agent (opcional)")
with c3:
    product = st.text_input("Produto (opcional)")

df = fetch_opportunities(deal_stage=deal_stage or None, sales_agent=sales_agent or None, product=product or None)

if df.empty:
    st.warning("Sem dados para os filtros selecionados.")
    st.stop()

# Apenas deals abertos para priorização
df = df[df["deal_stage"].isin(["Prospecting", "Engaging"])].copy()
df["days_in_pipeline"] = df["engage_date"].apply(days_since)
df["priority_score"] = df.apply(score_row, axis=1)
df["why"] = df.apply(explain_row, axis=1)
df["next_action"] = df.apply(next_action, axis=1)

top = df.sort_values("priority_score", ascending=False).head(10)

st.subheader("1) Quais deals atacar hoje?")
st.dataframe(
    top[
        [
            "opportunity_id",
            "sales_agent",
            "account",
            "product",
            "deal_stage",
            "days_in_pipeline",
            "priority_score",
        ]
    ],
    use_container_width=True,
)

st.subheader("2) Por que esses deals?")
for _, r in top.iterrows():
    st.write(f"- **{r['opportunity_id']}** ({r['account'] or 'sem conta'}): {r['why']}")

st.subheader("3) Qual a próxima ação recomendada?")
for _, r in top.iterrows():
    st.write(f"- **{r['opportunity_id']}**: {r['next_action']}")
```
